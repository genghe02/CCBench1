"""
This is a class for simulating the automatic playing of a guitar based on a given text representation of chords and tunes.
"""

class AutomaticGuitarSimulator:
    def __init__(self, text) -> None:
        """
        Initialize the AutomaticGuitarSimulator object with the given text.
        :param text: str, the text representation of chords and tunes to be played.
        """
        self.play_text = text

    def interpret(self, display=False):
        """
        Interpret the given text representation of chords and tunes and return a list of dictionaries containing the chord and tune information.
        :param display: bool, optional, if True, display the chord and tune information during the interpretation process.
        :return: list, a list of dictionaries containing the chord and tune information.
        """
        if len(self.play_text) == 0:
            return []
        else:
            play_list = []
            play_segs = self.play_text.split(" ")
            for play_seg in play_segs:
                pos = 0
                chord = ""
                for ele in play_seg:
                    if ele.isalpha():
                        pos += 1
                        chord += ele
                    else:
                        break
                play_list.append({'Chord': chord, 'Tune': play_seg[pos:]})
                if display:
                    self.display(chord, play_seg[pos:])
            return play_list

    def display(self, key, value):
        """
        Display the chord and tune information in a formatted string.
        :param key: str, the chord name.
        :param value: str, the tune value.
        :return: str, a formatted string containing the chord and tune information.
        """
        return "Automatic Guitar Playing -- Chord: %s, Play Tune: %s" % (key, value)
