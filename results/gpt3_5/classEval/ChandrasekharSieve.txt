    def generate_primes(self):
        <description for whole function>

        if self.n < 2:
            return []

        # Implementation of the Sieve of Eratosthenes algorithm to generate prime numbers up to self.n.
        sieve = [True] * (self.n + 1)
        sieve[0] = sieve[1] = False

        p = 2
        while p * p <= self.n:
            if sieve[p]:
                for i in range(p * p, self.n + 1, p):
                    sieve[i] = False
            p += 1

        primes = []
        for i in range(2, self.n + 1):
            if sieve[i]:
                primes.append(i)

        return primes
