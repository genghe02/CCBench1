        <description for all parameters>
        <description for return statement>
        <some test cases for the function>
        if m is None:
            m = len(self.datas)

    def select_all(self):
        <description for whole function>
        <description for all parameters>
        <description for return statement>
        <some test cases for the function>
        result = []
        for i in range(1, len(self.datas) + 1):

        return result

    @staticmethod
    def factorial(n):
        <description for whole function>
        <description for all parameters>
        <description for return statement>
        <some test cases for the function>
        result = 1
        for i in range(2, n + 1):
            result *= i
        return result
